第一題 以"從 1 累加到 100"

# 初始化總和變量
total_sum = 0

# 從 1 累加到 100
for i in range(1, 101):
    total_sum += i

# 顯示結果
print("從 1 累加到 100 的總和是:", total_sum)

第二題 "addto"

# 定義函式 addto
def addto(start, end):
    total_sum = 0
    for i in range(start, end + 1):
        total_sum += i
    return total_sum

# 呼叫函式 addto，並輸出結果
result = addto(1, 100)
print("從 1 累加到 100 的總和是:", result)

第三題 add_only_even

# 定義函式 add_only_even，接受兩個參數 start 和 end
def add_only_even(start, end):
    total_sum = 0
    # 使用 for 迴圈遍歷從 start 到 end 的所有數字
    for i in range(start, end + 1):
        if i % 2 == 0:  # 判斷是否為偶數
            total_sum += i  # 累加偶數
    return total_sum

# 讓用戶輸入起始值和結束值
start = int(input("請輸入起始整數: "))
end = int(input("請輸入結束整數: "))

# 呼叫函式並輸出結果
result = add_only_even(start, end)
print(f"從 {start} 到 {end} 的偶數總和是: {result}")

第四題 add_avoid_8

def add_avoid_8(start, end):
    total = 0
    for num in range(start, end + 1):
        if '8' not in str(num):  # 檢查數字是否包含 "8"
            total += num
    return total
 
# 呼叫函式，計算從 1 加到 100 並避免數字包含 "8" 的總和
total = add_avoid_8(1, 100)
 
# 印出計算結果
print(total)  # 輸出避開 "8" 的總和結果

第五題add_avoid_8 8跟3去掉

# 定義函式，計算不包含數字 "3" 和 "8" 的數字總和
def add_avoid_3_and_8(start, end):
    total = 0
    for num in range(start, end + 1):
        if '3' not in str(num) and '8' not in str(num):  # 檢查數字是否包含 "3" 或 "8"
            total += num
    return total

# 呼叫函式，計算範圍內的總和
start = 1
end = 100
total = add_avoid_3_and_8(start, end)

# 印出計算結果
print(f"從 {start} 到 {end} 的數字總和（排除包含 '3' 和 '8' 的數字）為：{total}")
